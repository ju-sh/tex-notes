\question
Express $\phi(N)$ in terms of $p$ and $q$.

\begin{solution}
  \begin{align*}
\phi(N) &= \phi(pq) \\
        &= \phi(p) \cdot \phi(q) \\
        &= (p-1)(q-1) \\
  \end{align*}
  
Because:
\begin{itemize}
\item $\phi(n) = n-1$ when $x$ is prime.
\item Chinese remainder theorem
\end{itemize}

\end{solution}
